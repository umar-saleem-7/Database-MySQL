1)
SELECT D.dname, E.ename, D.loc
FROM DEPT D, EMP E
WHERE E.deptno = D.deptno
AND E.comm IS NOT NULL;

2)
SELECT MOD(sal, 300) AS "Remainder"
FORM EMP
WHERE job = 'MANAGER';

3)
SELECT E.hiredate, E.comm, E.sal, D.dname, D.loc, S.grade
FROM EMP E, DEPT D, SALGRADE S
WHERE E.deptno = D.deptno
AND E.sal IN BETWEEM S.losal AND S.hisal;

4)
SELECT MAX(SAL) HIGHEST, MIN(SAL) LOWEST, SUM(SAL) SUM, ROUND(AVG(SAL)) AVERAGE, COUNT(EMPNO) "NIMBER OF ROWS"
FROM EMP;

5)
SELECT SUM(SAL) SUM, ROUND(AVG(SAL),2) AVERAGE
FROM EMP
WHERE DEPT = 10;

6)
SELECT D.DEPTNO, D.DNAME, D.LOC
FROM EMP E, DEPT D
WHERE E.DEPTNO = D.DEPTNO
GROUP BY D.DEPTNO, D.DNAME, D.LOC
HAVING MIN(E.SAL) > (SELECT AVG(SAL) FROM EMP);

7)
SELECT M.ENAME "MANAGERNAME", E.ENAME "EMPOLYEENAME"
FROM EMP E, EMP M
WHERE E.MGR = M.EMPNO;

8)
SELECT  ENAME, JOB, SAL
FROM EMP
WHERE SAL IN (SELECT MAX(SAL) FROM EMP GROUP BY DEPTNO);

9)
SELECT E.ENAME, E.JOB, E.SAL
FROM EMP E, EMP M
WHERE E.MGR = M.EMPNO
AND E.SAL > M.SAL;

10)
SELECT E.ENAME, E.HIREDATE, M.ENAME
FROM EMP E, EMP M
WHERE E.MGR= M.EMPNO
 AND E.HIREDATE > M.HIREDATE;
